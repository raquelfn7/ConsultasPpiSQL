SELECT tbPpiOficial.competencia, tbPpiOficial.atendimento, tbMunicipio.descricaoMunicipio, tbAssistencia.formaOrganizacao, tbFormaOrganizacao.descricaoFormaOrganizacao, tbBloco.descricaoBloco, tbSistema.descricaoSistema, tbComplexidade.descricaoComplexidade, Sum(IIf([financiamento]=6,[quantidadeProgramada],0)) AS qtPPI, Sum(IIf([FINANCIAMENTO]=6,[quantidadeProgramada]*([valorUnitarioFederal]+[valorUnitarioEstadual]),0)) AS valPPI, Sum(tbPpiOficial.valorFederalFES) AS valFES, Sum(IIf((([sadtFederal]+[sadtEstadual]))<>0,(([sadtFederal]+[sadtEstadual])),0)) AS sadt
FROM tbComplexidade INNER JOIN (((tbInstrumentoRegistro INNER JOIN ((((tbMunicipio INNER JOIN (tbOrigem INNER JOIN tbPpiOficial ON tbOrigem.idOrigem = tbPpiOficial.origem) ON tbMunicipio.idMunicipio = tbPpiOficial.atendimento) INNER JOIN tbAssistencia ON tbPpiOficial.assistencia = tbAssistencia.idAssistencia) INNER JOIN tbFormaOrganizacao ON tbAssistencia.formaOrganizacao = tbFormaOrganizacao.idFormaOrganizacao) INNER JOIN tbBloco ON tbPpiOficial.bloco = tbBloco.idBloco) ON tbInstrumentoRegistro.idInstrumentoRegistro = tbAssistencia.instrumentoRegistro) INNER JOIN tbSistema ON tbInstrumentoRegistro.sistema = tbSistema.idSistema) LEFT JOIN viSadt ON tbPpiOficial.idPpiOficial = viSadt.idPpiOficial) ON tbComplexidade.idComplexidade = tbAssistencia.complexidade
GROUP BY tbPpiOficial.competencia, tbPpiOficial.atendimento, tbMunicipio.descricaoMunicipio, tbAssistencia.formaOrganizacao, tbFormaOrganizacao.descricaoFormaOrganizacao, tbBloco.descricaoBloco, tbSistema.descricaoSistema, tbComplexidade.descricaoComplexidade, tbComplexidade.idComplexidade
HAVING (((tbComplexidade.idComplexidade)=3));
