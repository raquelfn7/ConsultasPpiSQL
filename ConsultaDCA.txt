FROM (tbMunicipio INNER JOIN (tbPpiOficial INNER JOIN tbAssistencia ON tbPpiOficial.assistencia = tbAssistencia.idAssistencia) ON tbMunicipio.idMunicipio = tbPpiOficial.atendimento) LEFT JOIN viSadt ON tbPpiOficial.idPpiOficial = viSadt.idPpiOficial
WHERE (((tbAssistencia.Subgrupo) In (907,908,909,910,917)))
GROUP BY tbPpiOficial.competencia, tbPpiOficial.atendimento, tbMunicipio.descricaoMunicipio, "SIHMC";
